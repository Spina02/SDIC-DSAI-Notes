\subsubsection{Prima Forma Fondamentale}

$$\begin{array}{ccc}
    Forma\ quadratica & \quad & Forma\ metrica
    \\[1em]
    ds^2 = E du^2 + 2F du dv + G dv^2 & \quad & g_{ij} = \begin{pmatrix}
    E & F \\
    F & G
\end{pmatrix}
\end{array}$$

\subsubsection*{Simboli di Christoffel}

$$
\begin{array}{ccc}
    Prima\ Specie & \qquad & Seconda\ Specie
    \\[1em]
    \displaystyle \Gamma_{ijk} = \frac{1}{2} \left( \frac{\partial g_{ik}}{\partial u^j} + \frac{\partial g_{ji}}{\partial u^k} - \frac{\partial g_{jk}}{\partial u^i} \right)
    & \quad & 
    \displaystyle \Gamma^i_{jk} = \frac{1}{2} g^{il} \left( \frac{\partial g_{lj}}{\partial u^k} + \frac{\partial g_{lk}}{\partial u^j} - \frac{\partial g_{jk}}{\partial u^l} \right)
\end{array}
$$

\subsubsection*{Equazione delle geodetiche}

\begin{align*}
    \frac{d^2u^i}{ds^2} + \Gamma^i_{jk} \frac{du^j}{ds} \frac{du^k}{ds} = 0
\end{align*}

\subsubsection*{Derivata covariante}

$$
A_{i,l} = \dfrac{DA_i}{\partial u^l} = \dfrac{\partial A_i}{\partial u^l} - \Gamma^m_{il} A_m
$$

R^i_{klm} = \dfrac{\partial \Gamma^i_{km}}{\partial u^l} - \dfrac{\partial \Gamma^i_{kl}}{\partial u^m} + \Gamma^n_{km} \Gamma^i_{nl} - \Gamma^n_{kl} \Gamma^i_{nm}
$$
