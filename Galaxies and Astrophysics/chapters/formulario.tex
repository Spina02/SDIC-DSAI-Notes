\subsubsection{Prima Forma Fondamentale}

\begin{itemize}
\item Forma quadratica:
$$
ds^2 = E du^2 + 2F du dv + G dv^2
$$

\item Forma metrica:
$$
g_{ij} = \begin{pmatrix}
    E & F \\
    F & G
\end{pmatrix}
$$
\end{itemize}

\subsubsection*{Simboli di Christoffel}

\begin{itemize}
\item Prima Specie:
    $$
    \Gamma_{ijk} = \frac{1}{2} \left( \frac{\partial g_{ik}}{\partial u^j} + \frac{\partial g_{ji}}{\partial u^k} - \frac{\partial g_{jk}}{\partial u^i} \right)
    $$
\item Seconda Specie:
    $$
    \Gamma^i_{jk} = \frac{1}{2} g^{il} \left( \frac{\partial g_{lj}}{\partial u^k} + \frac{\partial g_{lk}}{\partial u^j} - \frac{\partial g_{jk}}{\partial u^l} \right)
    $$
\end{itemize}

\subsubsection*{Equazione delle geodetiche}

\begin{align*}
    \frac{d^2u^i}{ds^2} + \Gamma^i_{jk} \frac{du^j}{ds} \frac{du^k}{ds} = 0
\end{align*}

\subsubsection*{Derivata covariante}

A_{i,l} = \dfrac{DA_i}{\partial u^l} = \dfrac{\partial A_i}{\partial u^l} - \Gamma^m_{il} A_m
